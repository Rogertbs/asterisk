#CONFIGURANDO DOIS RAMAIS PARA QUE POSSAM REALIZAR LIGAÇÕES ENTRE SI
#
#
#PRIMEIRAMENTE ACESSE FAÇA O BACKUP DO ARQUIVO sip.conf DA PASTA DO ASTERISK PARA
#QUE POSSA SER USADO PARA CONSULTA OU BACKUP.

sudo cp /etc/asterisk/sip.conf /etc/asterisk/sip.conf.bkp

#CRIE UM NOVO ARQUIVO sip.conf NO QUAL VAMOS EDITAR

sudo nano /etc/asterisk/sip.conf

#COMO PODEMOS PERCEBER O PROTOCOLO DE COMUNICAÇÃO QUE SERÁ USADO É O PROTOCOLO SIP,
#APÓS TERMINAR ESSE PROCESSO TEREMOS DOIS RAMAIS QUE PODERÃO SE COMUNICAR USANDO O
#PROTOCOLO SIP

#CONFIGURADO O ARQUIVO SIP.CONF
##DENTRO DO ARQUIVO SIP.CONF ESCREVA AS SEGUINTES LINHAS

[general]                     ;DECLARA VARIÁVIES GLOBAIS
udpbindaddr=0.0.0.0:5060      ;INDICA A FAIXA DE IP QUE O RAMAL ACEITA AUTENTICAÇÃO
context=ramais                ;CRIA UM CONTEXTO QUE SERÁ USADO NO PLANO DE DISCAGEM
disallow=all                  ;DESABILITA TODOS OS CODECS
allow=ulaw,alaw,gsm           ;HABILITA OS CODECS NECESSÁRIOS

[1000]                        ;INDICA O CONTEXTO, OU SEJA, NÚMERO DO RAMAL QUE SERÁ CONFIGURANDO
defaultuser=1000              ;USUÁRIO DE AUTENTICAÇÃO
type=friend                   ;TIPO FRIEND INDICA QUE ELE PODE FAZER E RECEBER LIGAÇÕES
secret=123456                 ;SENHA DO RAMAL, É RECOMENDADO USAR SENHA FORTES POR SEGURANÇA
dtmfmode=rfc2833              ;DEIXAR O PADRÃO
host=dynamic                  ;HOST DINAMICO
language=pt_BR                ;LINGUAGEM
allow=ulaw,alaw               ;CODECS USADOS PARA ESTE RAMAL

[1001]
defaultuser=1001
type=friend
secret=123456
dtmfmode=rfc2833
host=dynamic
language=pt_BR
allow=ulaw,alaw

#ASSIM APÓS APLICAR ESSAS CONFIGURAÇÕES NO ARQUIVO SIP.CONF SALVAR E SAIR DO ARQUIVO.
#É NECESÁRIO REGISTRAR OS RAMAIS NO ASTERISK USANDO O COMANDO ABAIXO:
#ACESSE A CLI DO ASTERISK E APÓS ISSO REGISTRE OS RAMAIS:

asterisk -vvvr

sip reload

sip show peers 
#mostra os ramais que foram cadastrados
